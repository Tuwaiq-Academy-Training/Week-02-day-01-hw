package com.example.springday01.controller;

import org.springframework.web.bind.annotation.*;

import java.util.ArrayList;

@RestController
public class employeeController {
    private ArrayList<String> list=new ArrayList<>();
// get user
    @GetMapping("employee")
    public ArrayList<String> getemployee () {
        return list;
    }

//post user
    @PostMapping("employee")
    public String addemployee (@RequestBody String employee ) {
        list.add(employee );

        return "add new employee";
    }
//edit user
    @PutMapping("employee/{index}")
    public String editemployee (@PathVariable int index,String employee ) {
        list.set(index, employee );
        return" Edit employee Successfully";
    }
    //delet user
    @DeleteMapping("employee/{index}")
    public String deletemployee (@PathVariable int index){
        list.remove(0);
        return "Employee Deleted Successfully";
    }}
